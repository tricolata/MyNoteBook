Install Kafka
    docker pull bitnami/kafka

Commands
    network
        docker network create app-tier --driver bridge
    zookeeper
        docker run -d --name zookeeper-server --network app-tier \
        -e ALLOW_ANONYMOUS_LOGIN=yes \
        bitnami/zookeeper:latest
    kafka server
        docker run -d --name kafka-server --network app-tier \
        -e ALLOW_PLAINTEXT_LISTENER=yes \
        -e KAFKA_CFG_ZOOKEEPER_CONNECT=zookeeper-server:2181 \
        bitnami/kafka:latest
    topics
        docker run -it --rm \
        --network app-tier \
        -e KAFKA_CFG_ZOOKEEPER_CONNECT=zookeeper-server:2181 \
        bitnami/kafka:latest 
        
        list - kafka-topics.sh --list  --bootstrap-server kafka-server:9092
        create - bin/kafka-topics.sh --create --topic topic1 --bootstrap-server localhost:9092
        describe - kafka-topics.sh --describe --topic quickstart-events --bootstrap-server localhost:9092
    producer
    consumer
        write - kafka-console-consumer.sh --topic quickstart-events --from-beginning --bootstrap-server localhost:9092
