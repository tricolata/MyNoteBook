add bin to path
	- temporary 
		edit $PATH	// only current terminal window
	- permanent
		edit .zshrc
		export PATH=$PATH:/Users/yourusername/bin
		source .zshrc	// refreshes terminal

	ex. intelliJ
		- nano /usr/local/bin/idea
			#!/bin/sh
			open -na "IntelliJ IDEA CE.app" --args "$@"
